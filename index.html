Clicking...") self.sb.scroll_to(result_link) time.sleep(random.uniform(1.5, 3.0)) try: self.sb.hover(result_link) time.sleep(0.5) self.sb.execute_script(f"document.querySelector('#search a[href*=\"{SITE_NAME}\"]').click();") except: self.sb.click(result_link) time.sleep(8) if SITE_NAME in self.sb.get_current_url(): print("🚀 Success! Landed on site.") return True self.sb.execute_script(f"window.scrollBy(0, {random.randint(400, 700)});") time.sleep(2) return False def perform_search(self, query): self.sb.uc_open_with_reconnect("https://www.google.com", 5) search_box = 'textarea[name="q"]' if self.sb.is_element_visible('textarea[name="q"]') else 'input[name="q"]' if self.sb.is_element_visible(search_box): print(f"🔍 Searching NEW Keyword: {query}") for char in query: self.sb.add_text(search_box, char) time.sleep(random.uniform(0.05, 0.25)) self.sb.add_text(search_box, "\n") time.sleep(random.randint(5, 8)) def engage(self): duration = random.randint(60, 120) # Increased stay time for new keywords print(f"⏱️ Staying for {duration}s...") start = time.time() while time.time() - start < duration: self.sb.execute_script(f"window.scrollBy(0, {random.randint(200, 600)});") time.sleep(random.randint(5, 15)) def run_session(i): query = random.choice(KEYWORDS) ua = random.choice(USER_AGENTS) res = random.choice(RESOLUTIONS) print(f"\n--- SESSION {i} ---") print(f"Keyword: {query} | Resolution: {res[0]}x{res[1]}") with SB(uc=True, test=True, agent=ua, incognito=True) as sb: sb.set_window_size(res[0], res[1]) agent = GSC_Agent(sb) agent.perform_search(query) success = agent.click_website_only() # If the new keyword doesn't show your site yet, use site: search to force it if not success: print(f"⏭️ Keyword '{query}' not ranking yet.
